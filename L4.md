#### Solución

En el presente laboratorio se genera una visualización de la solución propuesta para el problema 1 asignado, por medio del código creado en Python, que se muestra a continuación: 
# Se importa la librerÃ­a NumPy.
import numpy as np

# Se importa el mÃ³dulo stats de SciPy.
from scipy import stats

# Se importa la biblioteca "matplotlib" junto con la funciÃ³n "pyplot".
import matplotlib.pyplot as plt


# Variables aleatorias X y Y
va_X = stats.norm(0, np.sqrt(8))
va_Y = stats.norm(0, np.sqrt(8))

# CreaciÃ³n del vector de tiempo
T = 100			# nÃºmero de elementos
t_final = 10	# tiempo en segundos
t = np.linspace(0, t_final, T)

# InicializaciÃ³n del proceso aleatorio W(t) con N realizaciones
N = 100
W_t = np.empty((N, len(t)))	# N funciones del tiempo w(t) con T puntos

# CreaciÃ³n de las muestras del proceso w(t)
for i in range(N):
	X = va_X.rvs()
	Y = va_Y.rvs()
	w_t = X * np.cos(np.pi*t) + Y * np.sin(np.pi*t)
	W_t[i,:] = w_t
	plt.plot(t, w_t)

# Promedio de las N realizaciones en cada instante (cada punto en t)
P = [np.mean(W_t[:,i]) for i in range(len(t))]
plt.plot(t, P, lw=6, label='Valor esperado de 100 realizaciones')

# Graficar el resultado teÃ³rico del valor esperado
E = 0*t
plt.plot(t, E, '-.', lw=4, label='Valor esperado teÃ³rico')

# Mostrar las realizaciones del promedio calculado y teÃ³rico.
plt.title('Realizaciones del proceso aleatorio $W(t)$')
plt.xlabel('$t$')
plt.ylabel('$w_i(t)$')
plt.legend() # Se imprime las leyendas de la grÃ¡fica.
plt.show()   # Muestra la grÃ¡fica.

# T valores de desplazamiento tau.
desplazamiento = np.arange(T)
taus = desplazamiento/t_final

# InicializaciÃ³n de matriz de valores de correlaciÃ³n para las N funciones.
corr = np.empty((N, len(desplazamiento)))

# Nueva figura para la autocorrelaciÃ³n.
plt.figure()

# CÃ¡lculo de correlaciÃ³n para cada valor de tau.
for n in range(N):
	for i, tau in enumerate(desplazamiento):
		corr[n, i] = np.correlate(W_t[n,:], np.roll(W_t[n,:], tau))/T
	plt.plot(taus, corr[n,:])

# Valor teÃ³rico de autocorrelaciÃ³n.
Rww = np.sqrt(8) * np.cos(np.pi*taus)

# GrÃ¡ficas de autocorrelaciÃ³n para cada realizaciÃ³n del proceso.
plt.plot(taus, Rww, '-.', lw=4, label='AutocorrelaciÃ³n teÃ³rica')
plt.title('Funciones de autocorrelaciÃ³n de las realizaciones del proceso')
plt.xlabel(r'$\tau$')
plt.ylabel(r'$R_{WW}(\tau)$')
plt.legend() # Se imprime la leyenda de la grÃ¡fica.
plt.show()   # Muestra la grÃ¡fica.
